from datetime import date

def calculate_drill_date(today: date) -> str:
    """
    Returns drill_date as YYYY-MM-01 based on business rules
    """
    year = today.year
    month = today.month
    day = today.day

    # If today is on or after 20th → current month
    if day >= 20:
        drill_year = year
        drill_month = month
    else:
        # Before 20th → previous month
        if month == 1:
            drill_year = year - 1
            drill_month = 12
        else:
            drill_year = year
            drill_month = month - 1

    return f"{drill_year}-{drill_month:02d}-01"
